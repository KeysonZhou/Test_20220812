class Employee:
    def __init__(self,name,age):
        self.name=name
        self.age=int(age)
        self.ability=100-self.age
    def do_work(self):
        self.ability-=5
        return self.ability
    def strinfo(self):
        print(f'员工姓名：{self.name}   年龄：{self.age}   能力值：{self.ability}')
